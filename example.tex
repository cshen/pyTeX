# -*- coding: utf-8 -*-
#!/usr/bin/python

from sympy import latex
from sympy import *

x,y = symbols('x,y')

def is_eqal(exp1, exp2):
    exp1 = exp1.replace("$","")
    exp2 = exp2.replace("$","")
    return "$%s=%s$"%(exp1,exp2)

for el in ( (x+y)**2, (x+y)**3, (x+y)**4 ):
    print r"\item[$\bullet$] " + is_eqal(  latex(el), latex(el.expand()) )

\end{python}
